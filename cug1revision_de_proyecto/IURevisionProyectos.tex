\subsection{Pantalla:Revisión de proyectos}
\subsubsection{Objetivo}
Revisar el proyecto para cambiarlo de estado a ``Ejecución'' o ``Edición''.

\IUfig[0.6]{cu-g1/revision_proyectos.png}{IURevisionProyectos}{Revisión de Proyectos.}


\subsubsection{Comandos}
\begin{itemize}
 \item \IUbutton{Aceptar} Muestra mensaje de cambio de estado a ``Ejecución''.
 \item \IUbutton{Editar} Muestra mensade de cambio de estado a ``Edición''.

\end{itemize}
