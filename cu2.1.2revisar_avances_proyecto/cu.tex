	\begin{UseCase}{CU2.1.2}{Revisar Avances de Proyecto}{Permite ver las acciones registradas de un proyecto, el avance de las acciones del mismo, los presupuestos solicitados, aprobados y ejercidos en el dicho proyecto.}
		\UCitem{Versión}{3.5}
		\UCitem{Actor(es)}{Coordinador, Gerente}
		\UCitem{Propósito}{Contar con un mecanismo que ayude a revisar los avances de un proyecto.}
		\UCitem{Resumen}{Se muestran los datos de las acciones de un proyecto, su avance y presupuestos.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Datos del proyecto seleccionado.}
		\UCitem{Precondiciones}{\begin{itemize}
						\item Que se haya obtenido el perfil de Coordinador o el perfil de Gerente.
						\item Que exista al menos un proyecto registrado.
						\item Que el estado del proyecto sea Ejecución.
						\item Haber pasado por el \UCref{CU2.1}
					\end{itemize}
  					}
		\UCitem{Postcondiciones}{Ninguna.}
		%\UCitem{Autor}{Felipe López Hernández}
		\UCitem{Referencias}{CU-P2-061011.}
		\UCitem{Tipo}{Secundario. Viene de \UCref{CU2.1}}
		\UCitem{Módulo}{Coordinación.}
	\end{UseCase}

		\UCpaso Verifica que se cumpla la regla de negocio \BRref{RN70}.
