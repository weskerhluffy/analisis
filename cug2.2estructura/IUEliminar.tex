\subsection{Pantalla: Eliminar Tipo de Unidad}

\subsubsection{Objetivo}
Eliminar un Tipo de unidad seleccionado. Viene de \IUref{IUGestTipoDeUnidad}{Gestionar Tipo de Unidad.}

\subsubsection{Diseño}
\IUfig[0.6]{CU3/EliminarTipoDeUnidad2.jpg}{IUGestEliminarTipoDeUnidad}{Confirmación eliminar Tipo de Unidad.}

\subsubsection{Salidas}
Datos del registro seleccionado.

\subsubsection{Entradas}
Selección de Tipo de Unidad a eliminar.


\subsubsection{Comandos}
\begin{itemize}
 \item \IUbutton{Eliminar}:Elimina el registro del Tipo de unidad seleccionado y lleva a la pantalla \IUref{IUGestTipoDeUnidad}{Gestionar Tipo de Unidad.} 
 \item \IUbutton{Cancelar}: Regresa a la pantalla \IUref{IUGestTipoDeUnidad}{Gestionar Tipo de Unidad.}.
\end{itemize}

\subsubsection{Mensajes}
\begin{itemize}
	\item Ninguno
\end{itemize}
