	\begin{UseCase}{CU2.1.1}{Revisar Información de Proyecto}{Se muestra toda la información del proyecto, datos de responsable, datos de proyecto, así como la información de su alineación.}
		\UCitem{Versión}{1.0}
		\UCitem{Actor(es)}{Coordinador, Gerente.}
		\UCitem{Propósito}{Proporciona la información de un Proyecto registrado a detalle para revisión.}
		\UCitem{Resumen}{Revisar la información de un Proyecto registrado.}
		\UCitem{Entradas}{Ninguno.}
		\UCitem{Salidas}{Información de Proyecto \ref{dd:RevisarInformacionProyecto} .}
		\UCitem{Precondiciones}{Se presentan las siguientes precondiciones:
					\begin{itemize}
					 \item Caso de uso \UCref{CU2.1} con perfil Gerente o Coordinador.
					 \item Que exista un Proyecto registrado.
					\end{itemize}}
%		\UCitem{Postcondiciones}{Ninguna.}
		%\UCitem{Autor}{Adrian Martinez.}
		\UCitem{Referencias}{SIDAM-BESP-P1.}
		\UCitem{Tipo}{Secundario. Viene del caso de uso \UCref{CU2.1}}
		\UCitem{Módulo}{Coordinación, Gerencia.}	
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
			\UCpaso[\UCactor] Selecciona el proyecto dependiendo del perfil de usuario:
			\begin{itemize}
		    		\item El Coordinador selecciona el proyecto del \IUref{IUMenuCoordinador}{Menú del Coordinador}. % que se muestra de lado izquierdo en pantalla.
		  		\item El Gerente selecciona el proyecto del diagrama de gantt en la pantalla \IUref{IURevisarProyectoPerfilGerente}{Revisar Programa del perfil Gerente}.
			\end{itemize}
			\UCpaso[\UCactor] Selecciona la opción \IUbutton{Revisar Proyecto}.
			%\UCpaso[\UCactor] Elije el Proyecto del menu \IUref{IUMenuCoordinador}{Menú del Coordinador}. \Trayref{A}
			\UCpaso Obtiene los datos del Proyecto seleccionado.	\label{paso:CU2.1.1ObtieneDatos}
			\UCpaso Calcula las rutas de las alineaciones con los elementos EP1N. 
			\UCpaso Muestra la pantalla \IUref{IURevisarInformacionProyecto}{Revisar Información de Proyecto} con la información obtenida. \label{paso:CU2.1.1MuestraInformacion}
	\end{UCtrayectoria}
