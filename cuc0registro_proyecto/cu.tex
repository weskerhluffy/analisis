	\begin{UseCase}{CUC0}{Registrar Proyecto}{Consiste en ingresar los datos de un nuevo Proyecto, así como asociar Ejes Temáticos y Temas Transversales. Además, alinear el proyecto con un programa y/o con el programa sectorial.}
		\UCitem{Versión}{2.0}
		\UCitem{Actor(es)}{Coordinador}
		\UCitem{Propósito}{Editar los datos del registro del proyecto y establecer su alineacion.}
		\UCitem{Resumen}{Registrar los datos de un nuevo Proyecto.}
		\UCitem{Entradas}{
			\begin{itemize}
				\item Datos de Proyecto \ref{dd:DatosEditablesProyecto}.
				\item Seleccion de estructuras del programa sectorial.
				\item Seleccion de estructuras de un programa.
				\item Seleccion de ejes tematicos.
				\item Seleccion de temas transversales.
			\end{itemize}
		}
		\UCitem{Salidas}{Mensaje \ref{MSGCUC1}.}
		\UCitem{Precondiciones}{Se presentan las siguientes precondiciones
		  \begin{itemize}
		    \item Que existan programas registrados.
		    \item Que existan estructuras definidas para los programas.
		    \item Que existan Ejes Temáticos registrados.
		    \item Que existan Temas Transversales registrados.
		  \end{itemize}}
		\UCitem{Postcondiciones}{Ninguna.}
		%\UCitem{Autor}{Raúl Alberto Soto Cruz}
		\UCitem{Referencias}{Ninguna}
		\UCitem{Tipo}{Primario}
		\UCitem{Módulo}{Coordinación}	
	\end{UseCase}

			\UCpaso Muestra la pantalla \IUref{IURegistrarProyecto}{Registro de Proyecto}.

			\UCpaso Verifica que se cumpla la regla de negocio \BRref{RN33}.\Trayref{E}
			\UCpaso Verifica que se cumpla la regla de negocio \BRref{RN40}.\Trayref{F}

	\begin{UCtrayectoriaA}{A}{Cancelar operación}{El usuario abandona el Caso de Uso.}
			\UCpaso[\UCactor] Decide ya no editar los datos de un Proyecto.
			\UCpaso[\UCactor] Oprime el botón \IUbutton{Cancelar}.
			\UCpaso Muestra el menu \IUref{IUMenuCoordinador}{Menú del Coordinador}.
	\end{UCtrayectoriaA}


	\begin{UCtrayectoriaA}{B}{Datos del Proyecto Incorrectos}{Los datos de Proyecto proporcionados estan incompletos.}
			\UCpaso Muestra el mensaje (MSG-1) indicando que los datos ingresados estan incompletos. \ref{MSG1}
			\UCpaso Continúa en el paso \ref{paso:CUC0_IngresaDatosProyecto}.
	\end{UCtrayectoriaA}

	\end{UCtrayectoriaA}
