  \begin{UseCase}{CU2}{Revisar Programa}{Mostrar los proyectos, estructuras y diagrama de Gantt asociados a los diferentes programas registrados.}
    \UCitem{Versión}{1.0}
    \UCitem{Actor(es)}{Gerente, Secretario.}
    \UCitem{Propósito}{Se muestra la estructura del programa incluyendo los proyectos alineados en un sólo diagrama de Gantt.}
    \UCitem{Resumen}{Se muestran los Programas/Estructuras/Proyectos asociado al Gerente/Secretario con su diagrama de Gantt según el perfil de usuario con el que se haya accedido al sistema.}
    \UCitem{Entradas}{Ninguna.}
    \UCitem{Salidas}{Estructuras y programas asociados en un diagrama de Gantt con porcentaje de avances, alertas y el número de restricciones:
      \begin{itemize}
       \item Atendidas
       \item Pendientes
       \item Turnadas
      \end{itemize}}
    \UCitem{Precondiciones}{
         Haberse logueado con un perfil de Secretario y que exista al menos un programa registrado o haberse logueado con un perfil de Gerente y que exista al menos un programa asociado a éste perfil.}
    \UCitem{Postcondiciones}{Ninguna.}
    \UCitem{Autor}{Roberto Pérez Vargas.}
    \UCitem{Referencias}{}
    \UCitem{Tipo}{Primario.}
    \UCitem{Módulo}{Gerencia/Secretaría.}
  \end{UseCase}

  \begin{UCtrayectoria}{Principal}
    \UCpaso[\UCactor] Selecciona la opción \IUbutton{Revisar Programas} en el menú \IUref{IUMenuGerente}{Menú para Gerente.} \Trayref{A}. \label{paso:selecciona}
    \UCpaso Busca los Programas, Estructuras y Proyectos asociados al usuario.
    \UCpaso Cuenta las restricciones atendidas, pendientes y turnadas. \label{paso:Cuenta}
    \UCpaso Genera las alertas para cada proyecto.
    \UCpaso Construye la gráfica de Gantt.
    \UCpaso Muestra la pantalla: \label{paso:muestra}
    \begin{itemize}
     \item Para la Secretaría \IUref{IURevisarProgramaSecretaria}{Revisar Programa Secretaría}, con la posibilidad de \UCExtensionPoint{CUS1}{Gestionar Programas} y \UCExtensionPoint{CU2.1}{Revisar Proyecto} \Trayref{B} \Trayref{C}
     \item Para el Gerente \IUref{IURevisarProgramaGerente}{Revisar Programa Gerente}, con la posibilidad de \UCExtensionPoint{CU2.1}{Revisión Proyecto} y \UCExtensionPoint{CUG2}{Gestión del programa} \UCExtensionPoint{CUG2.2}{Gestionar Estructura}\Trayref{B} \Trayref{C}
    \end{itemize}
  \end{UCtrayectoria}
  
  \begin{UCtrayectoriaA}{A}{Inicio de sesión de un Gerente}{El usuario que inicia sesión es un Gerente}
    \UCpaso [\UCactor] Selecciona la opción \IUbutton{Revisar Programas} en el menú \IUref{IUMenuGerente}{Menú de gerente}.
    \UCpaso Carga automáticamente el programa asociado al gerente.
    \UCpaso Busca las Estructuras y Proyectos asociados al usuario.
    \UCpaso Continúa en el paso \ref{paso:Cuenta} de \UCref{CU2}.
  \end{UCtrayectoriaA}

  \begin{UCtrayectoriaA}{B}{Cancelar}{El usuario decide cancelar la operación}
    \UCpaso[\UCactor] Presiona el botón \IUbutton{Cancelar}
    \UCpaso Continúa en el paso \ref{paso:selecciona}. de \UCref{CU2}
  \end{UCtrayectoriaA}

  \begin{UCtrayectoriaA}{B}{Expandir elemento.}{El usuario puede expandir cada elemento del diagrama que tenga elementos hijos asociados.}
    \UCpaso[\UCactor] Solicita expandir un elemento.
    \UCpaso Busca los elementos hijos del elemento seleccionado para expandir.
    \UCpaso Actualiza el diagrama de Gantt con los elementos hijos del elemento seleccionado.
    \UCpaso Regresa al paso \ref{paso:muestra} de \UCref{CU2}.
  \end{UCtrayectoriaA}
